001    package gate.stanford;
002    
003    import java.io.Serializable;
004    
005    /**
006     * Simple class representing a single dependency relation.  The "target"
007     * is the Annotation ID of the dependent; the "type" is the dependency 
008     * tag (<a href="http://nlp.stanford.edu/software/parser-faq.shtml#c">the
009     * Stanford Parser documentation</a> contains links to the tagset</a>; for example,
010     * nsubj = "nominal subject", dobj = "direct object).
011     */
012    public class DependencyRelation implements Serializable {
013      /**
014       * Serial version UID.
015       */
016      private static final long serialVersionUID = -7842607116149222052L;
017    
018      /**
019       * The type of the dependency relation (det, amod, etc.).
020       */
021      private String type;
022      
023      /**
024       * The ID of the token that is the target of this relation.
025       */
026      private Integer targetId;
027      
028      public DependencyRelation(String type, Integer targetId) {
029        this.type = type;
030        this.targetId = targetId;
031      }
032    
033      /**
034       * Return the dependency tag (type).
035       * @return the dependency tag
036       */
037      public String getType() {
038        return type;
039      }
040    
041      /**
042       * Set the dependency tag.
043       * @param type dependency tag
044       */
045      public void setType(String type) {
046        this.type = type;
047      }
048    
049      /**
050       * Return the GATE Annotation ID of the dependent.
051       * @return the Annotation ID
052       */
053      public Integer getTargetId() {
054        return targetId;
055      }
056    
057      /**
058       * Set the Annotation ID of the dependent.
059       * @param targetId the Annotation ID
060       */
061      public void setTargetId(Integer targetId) {
062        this.targetId = targetId;
063      }
064      
065      /**
066       * Format the data structure for display.
067       * For example, if type is "dobj" and the dependent has Annotation ID 37,
068       * return the String "dobj(37)". 
069       */
070      public String toString() {
071        return type + "(" + targetId + ")";
072      }
073    }
